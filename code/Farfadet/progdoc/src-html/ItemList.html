001import java.util.HashMap;
002import java.util.*;
003
004/**
005 * Project Zuul
006 * Classe for Inventory of player'items
007 *
008 * @author Manon
009 * @version v1
010 */
011public class ItemList
012{
013    public HashMap <String, Item> aInventoryHM;
014
015    /**
016     * Constructeur par defaut de ItemList
017     */
018    public ItemList () {
019        this.aInventoryHM = new HashMap <String, Item> ();
020    } //constructeur par defaut ()
021
022    // ## Accesseurs ##
023    /**
024     * Accesseur pour avoir l'item
025     * 
026     * @return Item item quon voulait avoir
027     * @param pItem item en question
028     */
029    public Item getItem (final String pItem) {
030        return this.aInventoryHM.get(pItem);
031    } //getItem(.)
032    /**
033     * getItemList() : Permet de recuperer la liste complete
034     * @return aItems : Liste complète
035     */
036    public HashMap<String, Item> getItemList(){
037        return this.aInventoryHM;
038    }//getItemList
039    
040    /**
041     * Accesseur pour avoir la liste des items dans l'inventaire
042     * 
043     * @return String liste des items dans la HashMap
044     */
045    public String getItemsString(){
046        String vList = "";
047        Set <String> keys = this.aInventoryHM.keySet();
048        for(String vItem : keys){
049            vList += vItem+" ";
050        }
051        return vList;
052    } //getItemString()
053
054    /**
055     * Accesseur set de l'item
056     * 
057     * @param pName nom de l'item
058     * @param pItem type de l'item
059     */
060    public void setItem(final String pName, final Item pItem){
061        this.aInventoryHM.put(pName, pItem);
062    } //getItem(..)
063
064    /**
065     * Accesseur de la taille total de l'inventaire du joueur
066     * 
067     * @return int qui correspond au poids actuel de l'inventaire
068     */
069    public int getTotalWeight(){
070        int vWeight = 0;
071        Set <String> keys = this.aInventoryHM.keySet();
072        for(String item : keys){
073            vWeight = vWeight+aInventoryHM.get(item).getWeight();
074        }
075        return vWeight;
076    } //getTotalWeight()
077
078    // ## Methodes ##
079    public void removeItem(final String pName){
080        this.aInventoryHM.remove(pName);
081    } //removeItem(.)
082
083    /**
084     * Permet de savoir si un item existe
085     * @param pName Nom auquel est associé l'item
086     * @return boolean true si l'item est dans la liste
087     */
088    public boolean containsKey (final String pName){
089        return this.aInventoryHM.containsKey (pName);
090    } //containsKey(.)
091} // ItemList
