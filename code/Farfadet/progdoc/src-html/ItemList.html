001import java.util.*;
002
003/**
004 * Project Zuul
005 * Classe for Inventory of player'items
006 *
007 * @author Manon
008 * @version v1
009 */
010public class ItemList
011{
012    public HashMap <String, Item> aInventoryHM;
013
014    /**
015     * Constructeur par defaut de ItemList
016     */
017    public ItemList () {
018        this.aInventoryHM = new HashMap <String, Item> ();
019    } //constructeur par defaut ()
020
021    // ## Accesseurs ##
022    /**
023     * Accesseur pour avoir l'item
024     * 
025     * @return Item item who want
026     * @param pItem item 
027     */
028    public Item getItem (final String pItem) {
029        return this.aInventoryHM.get(pItem);
030    } //getItem(.)
031    
032    /**
033     * getItemList() : Permet de reccuperer la liste complete
034     * @return aItems : Liste complète
035     */
036    public HashMap<String, Item> getItemList(){
037        return this.aInventoryHM;
038    }//getItemList
039    
040    /**
041     * Accesseur pour avoir la liste des items dans l'inventaire
042     * 
043     * @return String liste des items dans la HashMap
044     */
045    public String getItemsString(){
046        if (aInventoryHM.isEmpty()) {
047            return "inventory is empty !";
048        }
049        StringBuilder vList = new StringBuilder("inventory : ");
050        Set <String> keys = this.aInventoryHM.keySet();
051        for(String vItem : keys){
052            vList.append(" " + vItem);
053        }
054        return vList.toString();
055    } //getItemString()
056
057    /**
058     * Accesseur set de l'item
059     * 
060     * @param pName nom de l'item
061     * @param pItem type de l'item
062     */
063    public void setItem (final String pName, final Item pItem){
064        this.aInventoryHM.put(pName, pItem);
065    } //getItem(..)
066
067    /**
068     * Accesseur de la taille total de l'inventaire du joueur
069     * 
070     * @return int qui correspond au poids actuel de l'inventaire
071     */
072    public int getTotalWeight(){
073        int vWeight = 0;
074        Set <String> keys = this.aInventoryHM.keySet();
075        for(String item : keys){
076            vWeight = vWeight + aInventoryHM.get(item).getWeight();
077        }
078        return vWeight;
079    } //getTotalWeight()
080
081    // ## Methodes ##
082    /**
083     * @param pName name of the object which want to remove
084     * @param pItem item which want to remove
085     */
086    public void removeItem (final String pName, final Item pItem){
087        this.aInventoryHM.remove(pName, pItem);
088    } //removeItem(..)
089
090    /**
091     * @param pName name of the object which want to add
092     * @param pItem item which want to remove
093     */
094    public void addItem (final String pName, final Item pItem) {
095        this.aInventoryHM.put(pName, pItem);
096    } //addItem(..)
097    
098    /**
099     * Permet de savoir si un item existe
100     * 
101     * @param pName Nom auquel est associé l'item
102     * @return boolean true si l'item est dans la liste
103     */
104    public boolean containsKey (final String pName){
105        return this.aInventoryHM.containsKey (pName);
106    } //containsKey(.)
107    
108    /**
109     * @return String which contains informations of the inventory
110     */
111    public String inventory () {
112        return this.getItemsString() + "\n" + "and total weight : " + this.getTotalWeight() + "\n";
113    }//inventory()
114} // ItemList
