001import java.util.Stack;

003/**

010public class Player
011{
012    private String aName;
013    private Room aCurrentRoom;
014    private int aPoidsMax = 19;
015    private Item aItem;
016    private ItemList aInventory;
017    private UserInterface aGui; 
018    public Stack <Room> aAntRoom; //pile des room precedentes
019
020    /**
021     * Construtor of the player
022     * 
023     * @param pNom nom du joueur
024     * @param pCurrentRoom room ou commence le joueur
025     */
026    public Player (final String pNom, final Room pCurrentRoom){
027        this.aName = pNom;
028        this.aCurrentRoom = pCurrentRoom;
029        this.aAntRoom = new Stack <Room> ();
030        this.aInventory = new ItemList ();
031    } //constructeur par defaut de player
032
033    // ## Accesseurs ##
034    /**
035     * Accesseur of the current room
036     * 
037     * @return String de la salle
038     */
039    public Room getCurrentRoom (){
040        return this.aCurrentRoom;
041    } //getCurrentRoom()
042
043    /**
044     * @return int which is the ttale weight of the inventory
045     */
046    public int getPoidMax () {
047        return this.aPoidsMax;
048    } //getPoidMax()
049
050    /**
051     * Accesseur for the player
052     * 
053     * @return String du nom
054     */
055    public String getName () {
056        return this.aName;
057    } //getName()
058
059    /**
060     * Accesseur set de la currentRoom
061     * 
062     * @param pRoom current room
063     */
064    public void setCurrentRoom (final Room pRoom){
065        this.aCurrentRoom = pRoom;
066    } //setCurrentRoom(.)
067
068    /**
069     * Accesseur de l'inventaire
070     * 
071     * @return ItemList liste de l'inventaire du player
072     */
073    public ItemList getInventory (){
074        return this.aInventory;
075    } //getInventory()
076
077    /**
078     * Accesseur set du gui
079     * 
080     * @param pUserInterface interface du joueur
081     */
082    public void setGui (final UserInterface pUserInterface){
083        this.aGui = pUserInterface;
084    } //setGui(.)
085
086    /**
087     * @return Stack of the precedent room
088     */
089    public Stack <Room> getStackRoom(){
090        return this.aAntRoom;
091    } //getStackRoom()
092
093    // ## Methodes ##
094    /**
095     * Methode for change room
096     * 
097     * @param pRoom pour indiquer la salle suivante
098     */
099    public void changeRoom (final Room pRoom){
100        this.aCurrentRoom = pRoom;
101        this.aGui.println (this.aCurrentRoom.getLongDescription ());
102        if (aCurrentRoom.getImageName() != null)
103            aGui.showImage(aCurrentRoom.getImageName());
104    } //changeRoom(.)
105
106    /**
107     * Methode for take a item in currentRoom
108     * 
109     * @param pCommand command of the item which you want to take
110     */
111    public void takeItem (final Command pCommand) {
112        String vDescr = pCommand.getSecondWord();
113        Item vItem = this.aCurrentRoom.aItemHM.get(vDescr);
114        if (vItem == null) 
115            aGui.println ("this object do not exist");
116        else if (this.aPoidsMax < this.aInventory.getTotalWeight() + vItem.getWeight())
117            aGui.println ("your inventory is too heavy");
118        else {
119            this.aInventory.addItem(vDescr, vItem);
120            this.aCurrentRoom.aItemHM.remove(vDescr, vItem);
121            this.aGui.println (this.aInventory.getItemsString());
122        }
123    } //takeItem(.)
124
125    /**
126     * Methode for drop a item in currentRoom
127     * 
128     * @param pCommand command of the item which you want to drop
129     */
130    public void dropItem (final Command pCommand) {
131        String vDescr = pCommand.getSecondWord();
132        Item vItem = this.aInventory.aInventoryHM.get(vDescr);
133        if (vItem == null)
134            aGui.println ("can't drop this object");
135        else {
136            aInventory.removeItem(vDescr, vItem);
137            this.aCurrentRoom.aItemHM.put(vDescr, vItem);
138        }
139        this.aGui.println (this.aInventory.getItemsString());
140    } //dropItem(.)     
141
142    /**
143     * Allow you to look around you
144     */
145    public void look () 
146    {
147        printLocationInfo();
148    } //look()
149
150    /**
151     * Allow you to eat something
152     * 
153     * @param pCommand what you want to eat
154     */
155    public void eat (final String pCommand)
156    {
157        Item vItem = this.aInventory.aInventoryHM.get(pCommand);
158        if (pCommand.equals("magicCookie") && this.aInventory.aInventoryHM.containsValue(vItem)) {
159            this.aPoidsMax = aPoidsMax*2;
160            this.aGui.println ("You eat the Magic Cookie, you increases inventory'weight");
161            aInventory.removeItem(pCommand, vItem);
162        }
163        else this.aGui.println ("You can't eat that");
164    } //eat(.) 
165
166    /**
167     * Allow you to go back
168     */
169    public void back () {
170        if ( ! this.aAntRoom.empty()) {
171            this.aCurrentRoom = this.aAntRoom.pop();
172            this.aGui.println (this.aCurrentRoom.getLongDescription());
173            if (this.aCurrentRoom.getImageName() != null) 
174                this.aGui.showImage (this.aCurrentRoom.getImageName());
175        }else
176            aGui.println ("--> You can't go back");        
177    } //back()
178
179    // ## Affichage ##
180    /**
181     * When you arrived in a new room, this methode print description and exits available
182     */
183    public void printLocationInfo ()
184    {
185        aGui.println (this.aCurrentRoom.getLongDescription());
186    } //printLocationInfo() 
187
188    /**
189     * The welcome text when you start a new game 
190     */
191    public void printWelcome ()
192    {
193        aGui.println("Welcome to the World of Zuul!");
194        aGui.println("The game name is : Alerte au pied de l'arc en ciel");
195        aGui.println("In a magical forest : you, a farfadet, are called urgently at the foot of the rainbow because gold coin from the cauldron have been stolen !");
196        aGui.println ("Then, you have to cross the forest to try to find the stolen gold coin");
197
198        aGui.println("Type 'help' if you need help.");
199        aGui.println("\n");
200
201        aGui.println(aCurrentRoom.getLongDescription()); 
202        aGui.showImage(aCurrentRoom.getImageName()); 
203    } //printWelcome()
204} //Player
