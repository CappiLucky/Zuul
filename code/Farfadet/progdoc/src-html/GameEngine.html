136    /**
137     * Given a command, process (that is: execute) the command.
138     * If this command ends the game, true is returned, otherwise false is
139     * returned.
140     * 
141     * @param pCommandLine text which was wirten by the user
142     */
143    public void interpretCommand(String pCommandLine) 
144    {
145        aGui.println ("*****************************************");
146        aGui.println("Your command is : " + pCommandLine);
147        Command command = aParser.getCommand(pCommandLine);
148        if(command.isUnknown()) {
149            aGui.println("I don't know what you mean...");
150            return;
151        }
152
153        String commandWord = command.getCommandWord();
154        if (commandWord.equals("help"))
155            printHelp();
156        else if (commandWord.equals("go"))
157            goRoom(command);
158        else if (commandWord.equals("quit")) {
159            if(command.hasSecondWord())
160                aGui.println("Quit what?");
161            else
162                endGame();
163        } 
164        else if (commandWord.equals("look"))
165            look();
166        else if (commandWord.equals("eat"))
167            eat();
168        else if (commandWord.equals("back"))
169            back();
170    }

184    /**
185     * Go in the room passed in parameters
186     * 
187     * @param pCommand The word which has writting
188     */
189    private void goRoom (final Command pCommand)
190    {
191        if(!pCommand.hasSecondWord()) {
192            // if there is no second word, we don't know where to go...
193            aGui.println("Go where?");
194            return;
195        }
196        
197        String vDirection = pCommand.getSecondWord();
198        
199        //try to leave current room
200        Room vNextRoom = aCurrentRoom.getExit(vDirection);
201
202        if (vNextRoom == null)
203        {
204            aGui.println ("There is no door !");
205        } else {
206            this.aAntRoom.push (this.aCurrentRoom);
207            this.aCurrentRoom = vNextRoom;
208            printLocationInfo();
209            if (aCurrentRoom.getImageName() != null)
210                aGui.showImage(aCurrentRoom.getImageName());
211        }    
212    } //procedure goRoom()
