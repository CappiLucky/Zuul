135    /**
136     * Given a command, process (that is: execute) the command.
137     * If this command ends the game, true is returned, otherwise false is
138     * returned.
139     * 
140     * @param pCommandLine text which was wirten by the user
141     */
142    public void interpretCommand(String pCommandLine) 
143    {
144        aGui.println ("*****************************************");
145        aGui.println("Your command is : " + pCommandLine);
146        Command command = aParser.getCommand(pCommandLine);
147        if(command.isUnknown()) {
148            aGui.println("I don't know what you mean...");
149            return;
150        }
151
152        String commandWord = command.getCommandWord();
153        if (commandWord.equals("help"))
154            printHelp();
155        else if (commandWord.equals("go"))
156            goRoom(command);
157        else if (commandWord.equals("quit")) {
158            if(command.hasSecondWord())
159                aGui.println("Quit what?");
160            else
161                endGame();
162        } 
163        else if (commandWord.equals("look"))
164            aPlayer.look();
165        else if (commandWord.equals("eat"))
166            if(!command.hasSecondWord())
167                aGui.println("Eat what?");
168            else aPlayer.eat(command.getSecondWord()); 
169        else if (commandWord.equals("back"))
170            aPlayer.back();
171        else if (commandWord.equals("test")) {
172            if (command.hasSecondWord())
173                test(command.getSecondWord());
174        }
175        else if (commandWord.equals("take"))
176            if (!command.hasSecondWord()) 
177                this.aGui.println ("Take what ? \n");
178            else 
179                aPlayer.takeItem(command);
180        else if (commandWord.equals("drop"))
181            aPlayer.dropItem(command);  
182        else if (commandWord.equals("inventory"))
183            this.aGui.println (this.aPlayer.getInventory().inventory());
184    }

199    /**
200     * Go in the room passed in parameters
201     * 
202     * @param pCommand The word which has writting
203     */
204    private void goRoom (final Command pCommand)
205    {
206        if(!pCommand.hasSecondWord()) {
207            // if there is no second word, we don't know where to go...
208            aGui.println("Go where?");
209            return;
210        }
211        String vDirection = pCommand.getSecondWord();
212        //try to leave current room
213        Room vNextRoom = aPlayer.getCurrentRoom().getExit(vDirection);
214        if (vNextRoom == null)
215        {
216            aGui.println ("There is no door !");
217            return;
218        }
219        //cas du trapDoor et lockDoor
220        Door vNextDoor = this.aPlayer.getCurrentRoom().getDoor(vDirection);
221        if (vNextDoor != null) {
222            if (this.aPlayer.getCurrentRoom().getDoor(vDirection).isTrapDoor()) {
223                if (! vNextDoor.canGo()) {
224                    aGui.println ("this door is a trap. You can't go in this direction");
225                    return;
226                } 
227                this.aPlayer.getStackRoom().clear();
228            } else if (this.aPlayer.getCurrentRoom().getDoor(vDirection).isLockDoor()) {
229                HashMap vInvtHM = this.aPlayer.getInventory().aInventoryHM;
230                if (vInvtHM.containsKey("goldCoin1") && vInvtHM.containsKey("goldCoin") && vInvtHM.containsKey("goldCoin3") && vInvtHM.containsKey("goldCoin4") && vInvtHM.containsKey("goldCoin5")) {
231                    vNextDoor.setLock(false);
232                    aGui.println ("Amazing ! You found a secret room...");
233                } else {
234                    aGui.println ("You can't open the door ! You have to take the 5 gold coin");
235                    return;
236                }
237            }
238        }
239
240        aPlayer.getStackRoom().push (aPlayer.getCurrentRoom());
241        aPlayer.changeRoom(vNextRoom);
242        if (aTimer == 0) {
243            aGui.println("!! time is over !!");
244            endGame();
245        }
246        else {
247            aTimer -= 1;
248            aGui.println("You have " + aTimer + "moves even");
249        }
250    } //procedure goRoom()

280    /**

282     */
283    private void test (final String pNom) {
284        Scanner vS;
285        try {
286            String vNom =pNom;
287            if (! vNom.endsWith(".txt")) vNom += ".txt";
288            InputStream vIPS = getClass().getResourceAsStream(vNom);
289            vS = new Scanner (vIPS);
290            while ( vS.hasNextLine() ) {
291                String vLine = vS.nextLine();
292                this.interpretCommand(vLine);
293            }
294        }
