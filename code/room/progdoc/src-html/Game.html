065    /**

070    private void goRoom (final Command pCommand)
071    {
072        String vDirection = pCommand.getSecondWord();
073        Room vNextRoom = null;
074        vNextRoom = aCurrentRoom.getExit(vDirection);
075
076        if (vNextRoom == null)
077        {
078            System.out.println ("There is no door !");
079        } else {
080            this.aCurrentRoom = vNextRoom;
081            printLocationInfo();
082        }    
083    } //procedure goRoom()

130    /**
131     * Methode which reconize the word written on the keyboard
132     * Available command "go" "help" "quit"
133     * 
134     * @param pCommand The word you written
135     * @return The result of the word 
136     */
137    private boolean processCommand (final Command pCommand)
138    {
139        if (pCommand.isUnknown())
140        {
141            System.out.println("I don't know what you mean...");
142            return false;
143        } 
144        String commandWord = pCommand.getCommandWord();
145        if (commandWord.equals("go"))
146        {
147            goRoom(pCommand);
148            return false;
149        } 
150        else if (commandWord.equals("help"))
151        {
152            printHelp();
153            return false;
154        } 
155        else if (commandWord.equals("quit")) 
156        {
157            return quit(pCommand);
158        } 
159        else if (commandWord.equals("look"))
160        {
161            look();
162            return false;
163        }
164        else if (commandWord.equals("eat"))
165        {
166            eat();
167            return false;
168        }
169        else return false;
170    } //processCommand(.)

172    /**
173     * The game engine
174     */
175    public void play()
176    {
177        Command vCommand;
178        printWelcome();
179        boolean vFinished = false;
180        while (vFinished == false)
181        {
182            vCommand = this.aParser.getCommand();
183            vFinished = this.processCommand(vCommand);
184        } 
185        System.out.println("Thank you for playing.  Good bye.");
186    }
